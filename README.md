```
npm install sequelize
npm install sqlite3
npm install --save-dev sequelize-cli
# 初期化
npx sequelize-cli init
# モデルとマイグレーションを作成
npx sequelize-cli model:generate --name Message --attributes text:string
# マイグレーション
npx sequelize-cli db:migrate
```
